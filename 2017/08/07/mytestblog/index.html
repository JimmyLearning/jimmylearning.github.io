dom4j解析中国天气网接口

**`

import java.util.List;

import org.dom4j.Attribute;
import org.dom4j.Document;
import org.dom4j.DocumentException;
import org.dom4j.Element;
import org.dom4j.io.SAXReader;

public class Dom4jGetWeather {
// 输入值为 省份拼音 和 对应省份下城市名的拼音 , 返回 查询的今日天气 字符串
// 如查询浙江杭州天气: String weather = Dom4jGetWeather.getWeather(“zhejiang”, “hangzhou”);
public static String getWeather(String province, String cityName) {
SAXReader read1 = new SAXReader();
String output = null;
try {
// 中国天气网xml接口
Document document = read1.read(“http://flash.weather.com.cn/wmaps/xml/“ + province + “.xml”);
Element rootElement = document.getRootElement();
// 拿到城市列表
List elements = rootElement.elements(“city”);
for (int i = 0; i < elements.size(); i++) {
Element thisElement = elements.get(i);
// 找到对应市级地区天气
Attribute attribute1 = thisElement.attribute(“pyName”);
if (attribute1.getValue().equals(cityName)) {
String city = thisElement.attribute(“cityname”).getValue();
String weather = thisElement.attribute(“stateDetailed”).getValue();
String lowtemp = thisElement.attribute(“tem1”).getValue();
String hightemp = thisElement.attribute(“tem2”).getValue();
// Attribute tempnow = thisElement.attribute(“temNow”);
String wind = thisElement.attribute(“windState”).getValue();
String humidity = thisElement.attribute(“humidity”).getValue();
// Attribute nowtime = thisElement.attribute(“time”);
// 输出格式：xx市今日天气为：雷阵雨转多云，28~38℃，湿度71%，西南风小于3级；
output = city + “市今日天气为：” + weather + “，” + lowtemp + “~” + hightemp + “℃，湿度” + humidity + “，”

                        + wind + "；";
            }
        }
    } catch (DocumentException e) {
        e.printStackTrace();
        throw new RuntimeException("输入的省份或城市拼音有误");
    }
    return output;
}

}
